void swap_var(int* a, int* b){

  int swap_a = *a;
  int swap_b = *b;

  cout << " swap values " << endl;

  *b = swap_a;
  *a = swap_b;

}

void swap_int(int & a, int& b){
  int hold;
  hold = a;
  a = b;
  b = hold;
}


double square (double x){ return x*x;}

int factorial (int x){
  int b = 1;
  for(int i = 1; i <= x; i++){
    b *= i;
  }
  return b;
}
